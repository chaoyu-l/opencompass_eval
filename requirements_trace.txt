# (Python 3.10.19, CUDA 12.1). To recreate it from scratch:
#
#     conda create -n opencompass python=3.10 -y
#     conda activate opencompass
#     pip install -r requirements_trace.txt
#     pip install -e .          # install this OpenCompass fork in editable mode
#
# Notes
#   * torch is pinned to a CUDA 12.1 wheel; the --extra-index-url line below is
#     required for pip to find it. On a different CUDA version (or CPU-only),
#     install a matching torch/torchvision yourself first, then run the command
#     above — pip keeps the torch you already have.
#   * flash-attn is intentionally NOT listed: the DREAM/LLaDA wrappers fall
#     back to eager attention when it is missing. Install it separately for FA2.
#   * Subjective evaluation (eval_*_subjective.py) needs extra packages — see
#     requirements/extra.txt (alpaca-eval, ...).
